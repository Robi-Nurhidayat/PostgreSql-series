######################## 216. Most Popular Users

menampilkan user yang Popular / paling banyak di tags

code:
select username, count(*)
from users
join(
	select user_id from photo_tags
	union all
	select user_id from caption_tags
) as tags on tags.user_id = users.id
group by username
order by count(*) desc
limit 5
